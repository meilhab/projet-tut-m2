\subsection{Implémentation du moteur de requ{\^e}tes}
\begin{frame}{Le langage NXQL}
	\begin{block}{Syntaxe de départ}
		SELECT * \\
		FROM <from-clause> \\
		[WHERE <where-clause>]
	\end{block}
	\begin{block}{Syntaxe d'arrivée}
		SELECT ( * | <select-clause> ) \\
		FROM <from-clause> \\
		[WHERE <where-clause>]
	\end{block}
\end{frame}

\begin{frame}{La grammaire}

	\centering
	\includegraphics[scale=0.45]{./images/depthfirstadapter}
	
	\begin{block}{Parcours d'arbre}
		\begin{itemize}
			\item Génération d'un ensemble de classes permettant de :
			\begin{itemize}
				\item construire un arbre syntaxique,
				\item parcourir cet arbre.
			\end{itemize}
			\item Implémentation du pattern visitor.
        \end{itemize}
	\end{block}
	
\end{frame}

\begin{frame}{La stratégie simple}

	\centering
	\includegraphics[scale=0.6]{./images/simpleQueryCompilator}
	
	\begin{block}{Concerne}
        \begin{itemize}
            \item Les requ{\^e}tes NXQL simples
            \item Transformation directe en critère ES
        \end{itemize}
    \end{block}	
	
   \begin{block}{Fonctionnement}
        \begin{itemize}
            \item Surcharger les méthodes \textit{out} du DepthFirstAdapter
            \item 1 critère NXQL <=> 1 critère ES
        \end{itemize}
    \end{block}
    
\end{frame}
