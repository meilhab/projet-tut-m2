\subsection{Transformation de modèles}
\begin{frame}{Méta modèle}
	\begin{block}{Signification}
		\begin{itemize}
			\item Grec \textit{méta} : changement, succession, transformation
			\item Niveau d'abstraction supérieur
			\item Un modèle représentant un modèle
		
		\end{itemize}			
	
	\end{block}

\end{frame}

\begin{frame}{Transformation de modèles}
	\centering
	\includegraphics[scale=0.5]{./images/exogene.png}

	\begin{block}{Passage d'un modèle à un autre}
		\begin{itemize}
			\item Modèle source conforme à son méta modèle
			\item Modèle cible conforme à son méta modèle
			\item Application de règles de transformation en utilisant les méta modèles source et cible
		
		\end{itemize}			
	
	\end{block}

\end{frame}

\subsection{Exemple fichier conforme au diagramme de séquence}
\begin{frame}{\'Ecriture d'un fichier d'entrée}
    \begin{block}{Explications}
        \begin{itemize}
            \item Fichier de type XMI (XML Metadata Interchange)
            \item Conforme au méta modèle du diagramme de séquence
            \item Contient deux séquences
            \item Une séquence reçoit/envoi des messages de/à l'extérieur 
            vers un composant appelé \textit{Environnement} 

        \end{itemize}

    \end{block}

\end{frame}
